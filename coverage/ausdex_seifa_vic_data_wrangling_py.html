36 vic_locations = load_victorian_suburbs_metadata()

49 ) 

50 # note, this was identifired

55 double_name_converter = { 

56 code: name

240def w_avg(df, values, weights): 

241 

242 d = df[values] 

243 w = df[weights]

245 out = (d * w).sum() / w.sum()
