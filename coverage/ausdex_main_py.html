129def seifa_vic( 

130 year_value: str, 

131 suburb: str, 

132 metric: Metric, 

133 lga: Union[str, None] = None, 

134 fill_value: str = "null", 

135 guess_misspelt: bool = False, 

136): 

137 """ 

138 Interpolates suburb aggregated socio-economic indexes for a given year for a given suburb. 

139 

140 Args: 

141 year_value (int, float, str): Year values in decimal years or in a string datetime format convertable by pandas.to_datetime function\n 

142 suburb (str): The name of the suburb that you want the data interpolated for\n 

143 metric (str): the name of the seifa_score variable, options are include\n 

144 `irsd_score` for index of relative socio-economic disadvantage,\n 

145 `ieo_score` for the index of education and opportunity,\n 

146 `ier_score` for an index of economic resources, `irsad_score` for index of socio-economic advantage and disadvantage,\n 

147 `uirsa_score` for the urban index of relative socio-economic advantage,\n 

148 `rirsa_score` for the rural index of relative socio-economic advantage\n 

149 lga (str None): local government area. Only necessary for suburb names that are repeated in the state\n 

150 fill_value (str): can be "extrapolate" to extrapolate past the extent of the dataset or "boundary_value" to use the closest datapoint, or \n 

151 or an excepted response for scipy.interpolate.interp1D fill_value keyword argument\n 

152 guess_misspelt (bool): If true, then if it cannot find a match for the name of the suburb then it finds the closest name to it. 

153 Returns: 

154 interpolated score (float) 

155 

156 """ 

157 import warnings 

158 

159 warnings.filterwarnings("ignore") 

160 

161 from .seifa_vic import interpolate_vic_suburb_seifa 

162 

163 result = interpolate_vic_suburb_seifa( 

164 year_value, 

165 suburb.upper(), 

166 metric.value, 

167 lga=lga, 

168 fill_value=fill_value, 

169 guess_misspelt=guess_misspelt, 

170 ) 

171 typer.echo(f"{result:.2f}")
